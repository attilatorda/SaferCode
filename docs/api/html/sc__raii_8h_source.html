4/*
5 * sc_raii.h
6 * RAII-style resource management and allocation helpers for SaferCode.
7 *
8 * Provides:
9 * - Unified allocation macros (SC_ALLOC / SC_REALLOC / SC_FREE) that swap
10 * in the sentinel allocator when SC_DEBUG_SENTINEL is defined.
11 * - Convenience type-safe allocation macros: NEW, NEW_ZERO, NEW_ARRAY, etc.
12 * - Checked variants (SAFE_NEW*) that abort on OOM with file:line context.
13 * - Scope-based cleanup via __attribute__((cleanup)) on GCC/Clang:
14 * AUTO_FREE_PTR, AUTO_FILE, AUTO_CHARBUF, AUTO_FD, AUTO_DIR
15 * - AUTODESTROY for typed struct pointers.
16 *
17 * Changes v1.4:
18 * - Added SC_REALLOC(ptr, sz) macro routed through sc_sentinel_realloc()
19 * in debug mode, plain realloc() in release mode. Fixes Issue #3.
20 *
21 * Usage:
22 * #include "sc_raii.h"
23 *
24 * void example(void) {
25 * AUTO_FREE_PTR char *buf = SC_ALLOC(256); // freed at scope exit
26 * AUTO_FILE FILE *f = fopen("x.txt", "r"); // closed at scope exit
27 * }
28 */

92static inline void* _sc_checked_alloc(size_t size, const char *file, int line) {
93 void *p = SC_ALLOC(size);
94 if (!p) {
95 char msg[256];
96 snprintf(msg, sizeof(msg),
97 "allocation failed at %s:%d (%zu bytes)",
98 file, line, size);
99 SC_PANIC(msg);
100 }
101 return p;
102}

118static inline void* _sc_checked_realloc(void *ptr, size_t size,
119 const char *file, int line) {
120 void *p = SC_REALLOC(ptr, size);

123 snprintf(msg, sizeof(msg),
