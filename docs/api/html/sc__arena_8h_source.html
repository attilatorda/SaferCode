4/*
5 * sc_arena.h
6 * Linear (bump) arena allocator for SaferCode.
7 *
8 * Features:
9 * - Two modes: dynamic (grows via SC_ALLOC) or fixed (user‑supplied buffer).
10 * - Allocations are bump‑pointer with configurable alignment.
11 * - Reset the arena to reuse memory without freeing.
12 * - Integrates with sc_raii.h (AUTO_ARENA, SC_ALLOC/SC_FREE).
13 * - Thread‑unsafe by design – caller must synchronise if needed.
14 *
15 * Usage:
16 * #include "sc_arena.h"
17 *
18 * // Dynamic arena (grows from heap)
19 * sc_arena_t arena;
20 * sc_arena_init(&arena, 4096); // initial capacity 4 KB
21 *
22 * void *p = sc_arena_alloc(&arena, 128);
23 * char *str = sc_arena_alloc_str(&arena, "hello");
24 *
25 * sc_arena_reset(&arena); // rewinds, memory can be reused
26 * sc_arena_destroy(&arena); // frees heap buffer
27 *
28 * // Fixed arena (user‑supplied buffer, no heap allocation)
29 * uint8_t buffer[2048];
30 * sc_arena_init_fixed(&arena, buffer, sizeof(buffer));
31 * // ... use arena, no need to destroy
32 */

80/* Initialise a dynamic arena with an initial capacity (bytes).
81 Capacity will be rounded up to alignment. Returns 0 on success, -1 on OOM. */
82static inline int sc_arena_init(sc_arena_t *arena, size_t initial_capacity) {
83 if (!arena) return -1;
84 if (initial_capacity == 0) initial_capacity = 64;
85 /* Align capacity to default alignment */
86 size_t align = SC_ARENA_DEFAULT_ALIGNMENT;
87 initial_capacity = (initial_capacity + align - 1) & ~(align - 1);
88
89 arena->buffer = (uint8_t*)SC_ALLOC(initial_capacity);
90 if (!arena->buffer) return -1;
91 arena->capacity = initial_capacity;
92 arena->offset = 0;
93 arena->type = SC_ARENA_DYNAMIC;
94 arena->owns_buffer = 1;
95 return 0;
96}

98/* Initialise a fixed arena from a user‑supplied buffer.
99 The buffer must remain valid for the lifetime of the arena.
100 No heap allocation occurs. */
101static inline void sc_arena_init_fixed(sc_arena_t *arena, void *buffer, size_t size) {
102 if (!arena || !buffer) return;
103 arena->buffer = (uint8_t*)buffer;
104 arena->capacity = size;
105 arena->offset = 0;

110/* Destroy an arena – frees dynamic buffer if owned.
111 After destroy, the arena must not be used unless re‑initialised. */
112static inline void sc_arena_destroy(sc_arena_t *arena) {
113 if (!arena) return;
114 if (arena->owns_buffer && arena->buffer) {
115 SC_FREE(arena->buffer);
116 }
117 arena->buffer = NULL;
118 arena->capacity = 0;
119 arena->offset = 0;
120}

122/* Reset the arena – rewinds the offset, allowing reuse of all memory.
123 Does not free the underlying buffer. */
124static inline void sc_arena_reset(sc_arena_t *arena) {
125 if (arena) arena->offset = 0;
126}

132/* Internal: align offset and check capacity.
133 Returns aligned offset, or SIZE_MAX if not enough space.
134 For dynamic arenas, we could grow here – we'll implement that later. */
135static inline size_t _sc_arena_align_offset(sc_arena_t *arena, size_t alignment) {
136 size_t off = arena->offset;
137 size_t aligned = (off + alignment - 1) & ~(alignment - 1);
138 return aligned;
139}

141/* Try to grow a dynamic arena when space is insufficient.
142 Returns 0 on success, -1 on failure. */
143static inline int _sc_arena_grow(sc_arena_t *arena, size_t min_capacity) {
144 if (arena->type != SC_ARENA_DYNAMIC) return -1; /* fixed cannot grow */
145
146#if SC_ARENA_NO_GROW
147 (void)min_capacity;
148 SC_PANIC("arena growth disabled (SC_ARENA_NO_GROW=1); "
149 "allocation would invalidate existing pointers");
150#else
151
152 size_t new_cap = arena->capacity;
153 do {
154 if (new_cap == 0) new_cap = 64;
155 else new_cap *= 2;
156 } while (new_cap < min_capacity);
157 /* Align new_cap to alignment */
158 size_t align = SC_ARENA_DEFAULT_ALIGNMENT;
159 new_cap = (new_cap + align - 1) & ~(align - 1);
160
161 uint8_t *new_buf = (uint8_t*)SC_ALLOC(new_cap);
162 if (!new_buf) return -1;
163 memcpy(new_buf, arena->buffer, arena->offset);
164 SC_FREE(arena->buffer);
165 arena->buffer = new_buf;
166 arena->capacity = new_cap;
167 return 0;
168#endif
169}

171/* Allocate a block of uninitialised memory from the arena.
172 Returns NULL if out of memory (or OOM on grow attempt). */
173static inline void* sc_arena_alloc(sc_arena_t *arena, size_t size) {
174 if (!arena || size == 0) return NULL;
175
176 size_t align = SC_ARENA_DEFAULT_ALIGNMENT;
177 size_t aligned_off = _sc_arena_align_offset(arena, align);
178 if (aligned_off + size > arena->capacity) {
179 /* Try to grow if dynamic */
180 if (arena->type == SC_ARENA_DYNAMIC) {
181 if (_sc_arena_grow(arena, aligned_off + size) != 0)
182 return NULL;
183 /* After grow, re‑compute aligned offset (offset unchanged, capacity increased) */
184 aligned_off = _sc_arena_align_offset(arena, align);
185 if (aligned_off + size > arena->capacity) return NULL; /* should not happen */
186 } else {
187 return NULL; /* fixed arena out of space */
188 }
189 }
190 void *ptr = arena->buffer + aligned_off;
191 arena->offset = aligned_off + size;
192 return ptr;
193}
