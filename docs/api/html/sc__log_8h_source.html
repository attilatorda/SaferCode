4/*
5 * sc_log.h
6 * Platform-independent logging for SaferCode.
7 *
8 * Targets:
9 * LOG_FILE -- append to a named file
10 * LOG_CONSOLE -- write to stdout
11 * LOG_NETWORK -- send over TCP (param: "host:port")
12 *
13 * Levels (in ascending severity):
14 * LOG_DEBUG (0) LOG_INFO (1) LOG_WARN (2) LOG_ERROR (3)
15 *
16 * Messages below the configured min_level are silently dropped.
17 * Every message is timestamped automatically.
18 *
19 * API:
20 * int log_init(log_target_t, log_level_t, const char *param)
21 * void log_message(log_level_t, const char *fmt, ...)
22 * int log_change(log_target_t, log_level_t, const char *param)
23 * void log_close(void)
24 *
25 * Fix (v1.1): LOG_NETWORK now uses inet_pton() instead of the deprecated
26 * inet_addr(). inet_addr() returns INADDR_NONE (== 255.255.255.255) on
27 * error, making error detection impossible for that address. inet_pton()
28 * has a distinct return value for errors and lays the groundwork for
29 * future IPv6 support.
30 *
31 * Note: this header is intentionally self-contained (header-only).
32 * Include it in exactly one translation unit per binary to avoid
33 * duplicate static-state issues across TUs.
34 */

84static int log_init(log_target_t target, log_level_t min_level, const char *param) {
85 _sc_log_close_internal();
86 g_sc_logger.target = target;
87 g_sc_logger.min_level = min_level;
88
89 if (target == LOG_FILE) {
90 if (!param) return -1;
91 g_sc_logger.file = fopen(param, "a");
92 return g_sc_logger.file ? 0 : -1;
93 }
94
95 if (target == LOG_NETWORK) {
96 if (!param) return -1;
97
98#ifdef _WIN32
99 WSADATA wsa;
100 if (WSAStartup(MAKEWORD(2, 2), &wsa) != 0) return -1;
101#endif
102 /* Parse "host:port" */
103 char host[256];
104 strncpy(host, param, sizeof(host) - 1);
105 host[sizeof(host) - 1] = '\0';
106
107 char *colon = strchr(host, ':');
108 if (!colon) return -1;
109 *colon = '\0';
110 int port = atoi(colon + 1);
111 if (port <= 0 || port > 65535) return -1;
112
113 g_sc_logger.sock = socket(AF_INET, SOCK_STREAM, 0);
114 if (g_sc_logger.sock == INVALID_SOCKET) return -1;
115
116 struct sockaddr_in addr;
117 memset(&addr, 0, sizeof(addr));
118 addr.sin_family = AF_INET;
119 addr.sin_port = htons((uint16_t)port);
120
121 /* Fix (v1.1): use inet_pton instead of deprecated inet_addr.
122 * inet_pton returns 1 on success, 0 for invalid address, -1 on error. */
123 int pton_result = inet_pton(AF_INET, host, &addr.sin_addr);
124 if (pton_result != 1) {
125 fprintf(stderr, "[SC_LOG] invalid host address: '%s'\n", host);
126 SC_CLOSE_SOCKET(g_sc_logger.sock);
127 g_sc_logger.sock = INVALID_SOCKET;
128 return -1;
129 }
130
131 if (connect(g_sc_logger.sock, (struct sockaddr*)&addr, sizeof(addr)) < 0) {
132 SC_CLOSE_SOCKET(g_sc_logger.sock);
133 g_sc_logger.sock = INVALID_SOCKET;
134 return -1;
135 }
136 }
137 return 0;
138}

140static void log_message(log_level_t level, const char *format, ...) {
141 if (level < g_sc_logger.min_level) return;
142
143 char buf[1024];
144 time_t now = time(NULL);
145 struct tm *t = localtime(&now);
146 int len = snprintf(buf, sizeof(buf),
147 "[%04d-%02d-%02d %02d:%02d:%02d] [%s] ",
148 t->tm_year + 1900, t->tm_mon + 1, t->tm_mday,
149 t->tm_hour, t->tm_min, t->tm_sec,
150 sc_level_strings[level]);
151
152 va_list args;
153 va_start(args, format);
154 vsnprintf(buf + len, sizeof(buf) - (size_t)len, format, args);
155 va_end(args);
156
157 strncat(buf, "\n", sizeof(buf) - strlen(buf) - 1);
158
159 switch (g_sc_logger.target) {
160 case LOG_CONSOLE:
161 fputs(buf, stdout);
162 fflush(stdout);
163 break;
164 case LOG_FILE:
165 if (g_sc_logger.file) {
166 fputs(buf, g_sc_logger.file);
167 fflush(g_sc_logger.file);
168 }
169 break;
170 case LOG_NETWORK:
171 if (g_sc_logger.sock != INVALID_SOCKET)
172 send(g_sc_logger.sock, buf, (int)strlen(buf), 0);
173 break;
174 }
175}
